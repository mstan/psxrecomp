# test_movement.txt — Boot → skip FMVs → menu → NEW GAME → test D-pad movement
#
# Strategy: same boot as test_new_game, then press RIGHT to see if Tomba walks.

turbo on
log === Boot sequence starting ===

# Hold CROSS through boot to skip FMVs
wait 10
hold CROSS

log Waiting for first state=4 (boot entry)...
wait_mem16 0x1FD848 4

log First state=4, waiting for fiber restart...
wait 5

wait_mem16 0x1FD848 0
log State dropped to 0 (fiber restarted)

wait_mem16 0x1FD848 4
log Second state=4 (main menu)!

release CROSS
wait 30

# Press CROSS for NEW GAME
log Pressing CROSS for NEW GAME...
press CROSS

wait 30

# Skip intro FMV with START presses
log Pressing START to skip NEW GAME FMV...
press START
wait 5
press START
wait 5
press START
wait 5
press START
wait 5
press START

# Wait for gameplay sub=1
log Waiting for sub=1 (gameplay dispatch)...
wait_mem16 0x1FD84A 1

log Gameplay reached!
wait 60

# Switch to normal speed for visual observation
turbo off
log Taking pre-movement screenshot...
screenshot 10_before_move.png

# Hold RIGHT for 120 frames (2 seconds) — Tomba should walk right
log Holding RIGHT for 2 seconds...
hold RIGHT
wait 120
screenshot 11_moving_right.png
release RIGHT
wait 30
screenshot 12_after_move.png

# Now try jumping — hold CROSS
log Pressing CROSS to jump...
press CROSS
wait 10
screenshot 13_jump.png

wait 60

# Try LEFT
log Holding LEFT for 2 seconds...
hold LEFT
wait 120
screenshot 14_moving_left.png
release LEFT

wait 30
screenshot 15_final.png
log === Movement test complete ===
exit 0
