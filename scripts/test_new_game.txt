#
# Strategy: hold CROSS through boot to skip all FMVs, press CROSS for NEW GAME,
# then press START once to skip the NEW GAME intro FMV, wait for gameplay.

turbo on
log === Boot sequence starting ===

# Hold CROSS through the entire boot sequence — skips FMV1, FMV2, and attract mode
wait 10
log Holding CROSS to skip all FMVs and attract mode...

# Wait for the first state=4 (boot entry's state machine)
log Waiting for first state=4 (boot entry)...
wait_mem16 0x1FD848 4

# Fiber restart happens here (0x800191E0 → 0x80019844).
# State drops to 0.  Keep holding CROSS — it will skip the attract FMV
# when state reaches 3, forcing state=4 immediately.
log First state=4 reached, waiting for fiber restart + second state=4...
wait 5

# Wait for state to drop (fiber restart sets state=0)
wait_mem16 0x1FD848 0
log State dropped to 0 (fiber restarted at FUN_80019844)

# Now wait for state=4 again — CROSS being held will skip attract FMV (state=3)
wait_mem16 0x1FD848 4
log Second state=4 reached (main menu)!

# Release CROSS and wait long enough for edge detection to clear
release CROSS
wait 30

# Verify we're at the interactive menu
turbo off
assert_mem16 0x1FD848 4 main_menu_interactive
log Main menu confirmed — taking screenshot
screenshot 01_main_menu.png

wait 30
log Pressing CROSS for NEW GAME...
